# compiled YARP.

cmake_minimum_required(VERSION 3.19)

find_package(YARP COMPONENTS os sig REQUIRED)

add_executable(framerate)
target_sources(framerate PRIVATE main.cpp)
target_link_libraries(framerate PRIVATE ${YARP_LIBRARIES})
