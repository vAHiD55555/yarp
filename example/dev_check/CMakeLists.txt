cmake_minimum_required(VERSION 3.19)
project(dev_check)

find_package(YARP COMPONENTS os sig dev REQUIRED)

add_executable(dev_check)
target_sources(dev_check PRIVATE main.cpp)
target_link_libraries(dev_check YARP::YARP_os
                                YARP::YARP_sig
                                YARP::YARP_dev
                                YARP::YARP_init)
