yarp_prepare_plugin(AudioRecorderWrapper
  CATEGORY device
  TYPE AudioRecorderWrapper
  INCLUDE AudioRecorderWrapper.h
  EXTRA_CONFIG
    WRAPPER=AudioRecorderWrapper
  DEFAULT ON
)

if(NOT SKIP_AudioRecorderWrapper)
  yarp_add_plugin(yarp_AudioRecorderWrapper)

  target_sources(yarp_AudioRecorderWrapper
    PRIVATE
      AudioRecorderWrapper.cpp
      AudioRecorderWrapper.h
  )

  target_link_libraries(yarp_AudioRecorderWrapper
    PRIVATE
      YARP::YARP_os
      YARP::YARP_sig
      YARP::YARP_dev
  )
  list(APPEND YARP_${YARP_PLUGIN_MASTER}_PRIVATE_DEPS
    YARP_os
    YARP_sig
    YARP_dev
  )

  yarp_install(
    TARGETS yarp_AudioRecorderWrapper
    EXPORT YARP_${YARP_PLUGIN_MASTER}
    COMPONENT ${YARP_PLUGIN_MASTER}
    LIBRARY DESTINATION ${YARP_DYNAMIC_PLUGINS_INSTALL_DIR}
    ARCHIVE DESTINATION ${YARP_STATIC_PLUGINS_INSTALL_DIR}
    YARP_INI DESTINATION ${YARP_PLUGIN_MANIFESTS_INSTALL_DIR}
  )

  set(YARP_${YARP_PLUGIN_MASTER}_PRIVATE_DEPS ${YARP_${YARP_PLUGIN_MASTER}_PRIVATE_DEPS} PARENT_SCOPE)

  set_property(TARGET yarp_AudioRecorderWrapper PROPERTY FOLDER "Plugins/Device")

  if(YARP_COMPILE_TESTS)
    add_subdirectory(tests)
  endif()

endif()
